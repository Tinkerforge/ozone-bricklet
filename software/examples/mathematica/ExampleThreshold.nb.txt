Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
oz=NETNew["Tinkerforge.BrickletOzone",uid,ipcon]
ipcon@Connect[host,port]

(* Get threshold callbacks with a debounce time of 10 seconds (10000ms) *)
oz@SetDebouncePeriod[10000]

(* Callback for ozone concentration greater than 20 ppb *)
OzoneConcentrationReachedCB[sender_,ozone_concentration_]:=
 Module[{},
  Print["Ozone Concentration: "<>ToString[N[Quantity[ozone_concentration,"ppb"]]]<>"."];
 ]

AddEventHandler[oz@OzoneConcentrationReached,OzoneConcentrationReachedCB]

(* Configure threshold for "greater than 20 ppb" *)
option=Tinkerforge`BrickletOzone`THRESHOLDUOPTIONUGREATER
oz@SetOzoneConcentrationCallbackThreshold[option,20,0]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[oz]
ReleaseNETObject[ipcon]
