Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change XYZ to the UID of your Ozone Bricklet*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
o=NETNew["Tinkerforge.BrickletOzone",uid,ipcon]
ipcon@Connect[host,port]

(*Get threshold callbacks with a debounce time of 10 seconds (10000ms)*)
o@SetDebouncePeriod[10000]

(*Callback function for ozone concentration reached callback (parameter has unit ppb)*)
OzoneConcentrationReachedCB[sender_,ozoneConcentration_]:=
 Print["Ozone Concentration: "<>ToString[N[Quantity[ozoneConcentration,"ppb"]]]]
AddEventHandler[o@OzoneConcentrationReached,OzoneConcentrationReachedCB]

(*Configure threshold for ozone concentration "greater than 20 ppb" (unit is ppb)*)
option=Tinkerforge`BrickletOzone`THRESHOLDUOPTIONUGREATER
o@SetOzoneConcentrationCallbackThreshold[option,20,0]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[o]
ReleaseNETObject[ipcon]
