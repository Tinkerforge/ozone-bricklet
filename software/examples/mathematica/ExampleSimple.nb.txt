Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="hbo"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
oz=NETNew["Tinkerforge.BrickletOzone",uid,ipcon]
ipcon@Connect[host,port]

(* Get current ozone concentration (unit is ppb) *)
Print["Ozone Concentration: "<>
      ToString[N[Quantity[oz@GetOzoneConcentration[],"ppb"]]]]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[oz]
ReleaseNETObject[ipcon]
