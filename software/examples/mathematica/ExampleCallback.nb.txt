Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="XYZ"(*Change to your UID*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
o=NETNew["Tinkerforge.BrickletOzone",uid,ipcon]
ipcon@Connect[host,port]

(*Callback function for ozone concentration callback (parameter has unit ppb)*)
OzoneConcentrationCB[sender_,ozoneConcentration_]:=
 Print["Ozone Concentration: "<>ToString[N[Quantity[ozoneConcentration,"ppb"]]]]
AddEventHandler[o@OzoneConcentration,OzoneConcentrationCB]

(*Set period for ozone concentration callback to 1s (1000ms)*)
(*Note: The ozone concentration callback is only called every second*)
(*if the ozone concentration has changed since the last call!*)
o@SetOzoneConcentrationCallbackPeriod[1000]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[o]
ReleaseNETObject[ipcon]
