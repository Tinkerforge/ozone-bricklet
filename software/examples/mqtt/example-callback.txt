# Change XYZ to the UID of your Ozone Bricklet

setup:
	# Handle incoming ozone concentration callbacks
	subscribe to tinkerforge/callback/ozone_bricklet/XYZ/ozone_concentration
	publish '{"register": true}' to tinkerforge/register/ozone_bricklet/XYZ/ozone_concentration # Register ozone_concentration callback
	
	# Set period for ozone concentration callback to 1s (1000ms)
	# Note: The ozone concentration callback is only called every second
	#       if the ozone concentration has changed since the last call!
	publish '{"period": 1000}' to tinkerforge/request/ozone_bricklet/XYZ/set_ozone_concentration_callback_period
