# Change XYZ to the UID of your Ozone Bricklet

setup:
	# Get threshold callbacks with a debounce time of 10 seconds (10000ms)
	publish '{"debounce": 10000}' to tinkerforge/request/ozone_bricklet/XYZ/set_debounce_period
	
	# Handle incoming ozone concentration reached callbacks
	subscribe to tinkerforge/callback/ozone_bricklet/XYZ/ozone_concentration_reached
	publish '{"register": true}' to tinkerforge/register/ozone_bricklet/XYZ/ozone_concentration_reached # Register ozone_concentration_reached callback
	
	# Configure threshold for ozone concentration "greater than 20 ppb"
	publish '{"option": "greater", "min": 20, "max": 0}' to tinkerforge/request/ozone_bricklet/XYZ/set_ozone_concentration_callback_threshold
